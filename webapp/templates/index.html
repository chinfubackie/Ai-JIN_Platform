⚡ Ai-JIN Inference API

    
ใช้งานโมเดลและเรียกใช้ผ่าน REST API — รองรับ GPU acceleration

    

      
      
      
    


    


import requests
import base64

# อ่านไฟล์ภาพและแปลงเป็น base64
with open("image.jpg", "rb") as f:
    img_b64 = base64.b64encode(f.read()).decode()

# ส่งไป inference
url = "http://localhost:8111"
response = requests.post(url, json={
    "command": "predict",
    "image": img_b64,
    "conf": 0.25,
    "iou": 0.45,
    "imgsz": 640
})

result = response.json()
print(f"Detected {result['count']} objects")
for det in result["detections"]:
    print(f"  {det['class_name']}: {det['confidence']:.2f} @ {det['bbox']}")

    


    


# Single image inference
curl -X POST http://localhost:8111 \
  -H "Content-Type: application/json" \
  -d '{
    "command": "predict",
    "image_path": "/dataset/auto_improve/images/train/JBA23-000030/img001.jpg",
    "conf": 0.25
  }'

# Start training
curl -X POST http://localhost:8111 \
  -H "Content-Type: application/json" \
  -d '{
    "command": "train",
    "config": {
      "data": "/dataset/auto_improve/data.yaml",
      "epochs": 100,
      "imgsz": 640,
      "batch": 16
    }
  }'

# Export model
curl -X POST http://localhost:8111 \
  -H "Content-Type: application/json" \
  -d '{
    "command": "export",
    "model": "/runs/train/weights/best.pt",
    "format": "onnx"
  }'

📘 API Endpoints

    

      
เมธอดEndpointคำอธิบาย

      
        
GET/healthสถานะเซิร์ฟเวอร์, ข้อมูล GPU, สถานะการเทรน

        
GET/modelsแสดงรายการโมเดลที่เทรนแล้ว

        
POST/ command=predictตรวจจับภาพเดียว (base64 หรือ path)

        
POST/ command=batch_predictตรวจจับหลายภาพในโฟลเดอร์

        
POST/ command=trainเริ่มงานเทรน

        
POST/ command=valรัน validation (mAP metrics)

        
POST/ command=exportส่งออกโมเดล (ONNX, TensorRT ฯลฯ)
